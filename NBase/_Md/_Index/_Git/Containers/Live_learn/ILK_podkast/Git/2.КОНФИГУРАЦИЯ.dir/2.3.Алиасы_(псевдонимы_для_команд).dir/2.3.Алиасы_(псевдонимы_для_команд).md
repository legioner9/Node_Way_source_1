## git alias 

### sis aliases

local

`git config alias.echoagf '!echo "agf"'`

    git echoagf
    <>agf

    [alias]
        ls = !ls
        echoagf = !echo \"agf\"



global 

`git config --global alias.echocrg '!echo "crg"'`
    git echocrg
    <>crg

### simple aliases

`git config --global alias.s 'status'`
